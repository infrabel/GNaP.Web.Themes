<div ng-controller="gnap-stacked-progressbar as vm">
    <input ng-repeat="bar in vm.collection track by $index" type="text" ng-model="bar.value">
    <hr>
    <div progress class="progress-small progress-striped active">
        <div bar ng-repeat="bar in vm.collection track by $index" value="bar.value" type="{{bar.type}}">
            <span>{{bar.value}}%</span>
        </div>
    </div>
</div>
(function () {
    angular
        .module('gnap-example-app')
        .controller('gnap-stacked-progressbar', GnapStackedProgressbarController);

    GnapStackedProgressbarController.$inject = [];

    function GnapStackedProgressbarController() {
        var vm = this;

        vm.collection = [];

        vm.collection.push({
            value: 20,
            type: 'success'
        });

        vm.collection.push({
            value: 15,
            type: 'warning'
        });

        vm.collection.push({
            value: 25,
            type: 'error'
        });
    };
})();
