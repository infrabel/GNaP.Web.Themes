<div class="form-group" ng-controller="gnap-date-range-picker as vm">
    <label class="col-sm-3 control-label no-padding-right">Date Range</label>
    <div class="col-sm-9">
        <span gnap-daterange date-begin="vm.range.dateStart" date-end="vm.range.dateEnd" icon-position="{{vm.iconPosition}}"></span>
    </div>
</div>
(function () {
    angular
        .module('gnap-example-app')
        .controller('gnap-date-range-picker', GnapDaterangepickerController);

    GnapDaterangepickerController.$inject = [];

    function GnapDaterangepickerController() {
        var vm = this;

        vm.range = {
            dateStart: new Date(2014, 0, 7),
            dateEnd: new Date(2014, 7, 14)
        };

        vm.iconPosition = 'left';
    };
})();
