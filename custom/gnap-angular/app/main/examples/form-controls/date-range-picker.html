<div class="form-group" ng-controller="gnap-date-range-picker">
    <label class="col-sm-3 control-label no-padding-right">Date Range</label>
    <div class="col-sm-9">
        <span gnap-date-range date-begin="range.dateStart" date-end="range.dateEnd" icon-position="{{iconPosition}}"></span>
    </div>
</div>
(function () {
    angular
        .module('gnap-example-app')
        .controller('gnap-date-range-picker', GnapDaterangepickerController);

    GnapDaterangepickerController.$inject = ['$scope'];

    function GnapDaterangepickerController($scope) {
        $scope.range = {
            dateStart: new Date(2014, 0, 7),
            dateEnd: new Date(2014, 7, 14)
        };

        $scope.iconPosition = 'left';
    };
})();
