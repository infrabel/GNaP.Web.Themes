<div ng-controller="gnap-confirm-dialog">
    <button class="btn btn-primary"
            ng-bootbox-confirm="Are you sure you want to confirm this?"
            ng-bootbox-confirm-action="confirm()"
            ng-bootbox-confirm-action-cancel="cancel()">
        Click Me!
    </button>
</div>
angular
    .module('gnap-example-app')
    .controller('gnap-confirm-dialog', function($scope) {
        $scope.confirm = function() {
            alert('You were sure!');
        };

        $scope.cancel = function() {
            alert('You weren\'t sure!');
        };
    });
