<div ng-controller="gnap-progressbar">                        
  <div class="form-group">
    <label class="control-label">Progress</label>
    <div><input type="text" ng-model="progress"></div>
  </div>
  <div progressbar class="progress-small progress-striped active" value="progress"></div>
  <div progressbar class="progress-small progress-striped" value="workLeft"></div>
</div>
angular
.module('gnap-app')
  .controller('gnap-progressbar', function($scope) {
    $scope.progress = 40;
    $scope.$watch('progress', function(progress) { $scope.workLeft = 100 - progress; });
  });
